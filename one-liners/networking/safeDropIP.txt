safedropip(){ comment="Remove if customer complains."; [[ $2 ]] && comment=$2; iptablecnt=$(iptables -nL | grep $1 | wc -l); commonsafecnt=$(whois $1 | grep -iP '(google|bing|cloudflare|sucuri|websitewelcome)' | wc -l); [[ $(( $iptablecnt + $commonsafecnt )) -lt 1 ]] && iptables -I INPUT -j DROP -m comment --comment "'$comment -bdupree $(date +%D)'" -s $1 || echo "$1 Failed. $iptablecnt-$commonsafecnt"; }; safedropip

 mysql -uadmin -p`cat /etc/psa/.psa.shadow` -e "SELECT domains.name as Name, hosting.www_root AS 'Document Root' FROM domains LEFT JOIN hosting on domains.id=hosting.dom_id WHERE domains.htype='vrt_hst' UNION SELECT CONCAT(subdomains.name, '.', domains.name) AS Name, subdomains.www_root AS 'Document Root' FROM domains INNER JOIN subdomains ON domains.id=subdomains.dom_id LEFT JOIN hosting ON domains.id=hosting.dom_id WHERE domains.htype='vrt_hst';" psa

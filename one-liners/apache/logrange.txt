toStamp(){ nd=$1; [[ "$1" =~ [0-9]*/[A-Z][a-z]*/([0-9]*:){2}[0-9]* ]] && nd=$(echo $1 |  awk -F'[/:]' '{print $2" "$1" "$3,$4":"$5":"$6}'); date --date="$nd" +%s;}; logRange(){ starts=$(toStamp "$2"); ends=$(toStamp "$3"); awk -F'[ []+' '{print $4, $0}' $1 | while read ldate line; lstamp=$(toStamp $ldate); do [[ $lstamp > $starts && $lstamp < $ends ]] && echo $line; done;}; logRange
